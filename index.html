What the JavaScript is doing

// Step 1: When you copy from sourceBox → save it
sourceBox.addEventListener('copy', function() {
    allowedText = selected text from sourceBox
})

// Step 2: When you paste into targetBox → check it
targetBox.addEventListener('paste', function(e) {
    if (pastedText !== allowedText) {
        e.preventDefault()  // BLOCK the paste
    }
})
